#include "base.h"
#define DYN_ARRAY_MIN_CAP 16

typedef struct {
    $type* v;
    size_t len;
    size_t cap;
} $arrayType;

static inline
void $(prefix)_init($arrayType* arr, size_t len, size_t cap){
    arr->len = len;
    arr->cap = cap;
    arr->v = ($type*)calloc(cap, sizeof($type));
}

static inline
void $(prefix)_resize($arrayType* arr, size_t new_cap){
    bool requires_zeroing_excess = new_cap > arr->cap;

    arr->v = ($type*)realloc(arr->v, sizeof($type) * new_cap);
    ensure(arr->v, "failed to reallocate");

    arr->len = min(arr->len, new_cap);
    arr->cap = new_cap;

    if(requires_zeroing_excess){
        memset(&arr->v[arr->len], 0, sizeof($type) * (arr->cap - arr->len));
    }
}

static inline
void $(prefix)_clear($arrayType* arr){
    arr->len = 0;
}

static inline
void $(prefix)_append($arrayType* arr, $type value){
    if(arr->len >= arr->cap){
        size_t new_cap = max(arr->cap * 2, DYN_ARRAY_MIN_CAP);
        $(prefix)_resize(arr, new_cap);
    }

    arr->v[arr->len] = value;
    arr->len += 1;
}

static inline
bool $(prefix)_pop($arrayType* arr){
    if(arr->len == 0){
        return false;
    }

    arr->len -= 1;
    return true;
}

static inline
bool $(prefix)_pop_into($arrayType* arr, $type* out){
    if(arr->len == 0){
        return false;
    }

    arr->len -= 1;
    *out = arr->v[arr->len];
    return true;
}

static inline
void $(prefix)_insert($arrayType* arr, size_t idx, $type val){
    if(arr->len >= arr->cap){
        size_t new_cap = max(arr->cap * 2, DYN_ARRAY_MIN_CAP);
        $(prefix)_resize(arr, new_cap);
    }

    memmove(&arr->v[idx + 1], &arr->v[idx], sizeof($type) * (arr->len - 0));
    arr->v[idx] = val;
}
